1.Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
{
   let hrsangle=((h*360)/12+(m*360)/(12*60));
   let minangle=((m*360)/60);
   let angle=Math.abs(hrsangle-minangle);
   if (angle>180){
     angle=360-angle;
   }
   return angle;
};

3.Perfect Number Check.

        
var Perfect_Check = (N) => 
{
   let count=0;
   for(let i=1;i<=N/2;i++)
   {
     if(N%i===0)
     count +=i;
   }
   if(count==N && count < N)
   return "YES";
   else 
   return "NO";
        
};
